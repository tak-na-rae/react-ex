import React, { useState, useEffect } from 'react';

const SyncedInputs = () => {
	//입력값이 변경될 때마다 콘솔에 로그를 찍는 코드 구현

  return (
    <div>
	    {/* input1이 변경되면 input2도 변경되도록 작성 */}

};
